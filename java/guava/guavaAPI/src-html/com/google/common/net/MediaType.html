050/**
051 * Represents an <a href="http://en.wikipedia.org/wiki/Internet_media_type">Internet Media Type</a>
052 * (also known as a MIME Type or Content Type). This class also supports the concept of media ranges
053 * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1">defined by HTTP/1.1</a>.
054 * As such, the {@code *} character is treated as a wildcard and is used to represent any acceptable
055 * type or subtype value. A media type may not have wildcard type with a declared subtype. The
056 * {@code *} character has no special meaning as part of a parameter. All values for type, subtype,
057 * parameter attributes or parameter values must be valid according to RFCs <a
058 * href="http://www.ietf.org/rfc/rfc2045.txt">2045</a> and <a
059 * href="http://www.ietf.org/rfc/rfc2046.txt">2046</a>.
060 *
061 * <p>All portions of the media type that are case-insensitive (type, subtype, parameter attributes)
062 * are normalized to lowercase. The value of the {@code charset} parameter is normalized to
063 * lowercase, but all others are left as-is.
064 *
065 * <p>Note that this specifically does <strong>not</strong> represent the value of the MIME {@code
066 * Content-Type} header and as such has no support for header-specific considerations such as line
067 * folding and comments.
068 *
069 * <p>For media types that take a charset the predefined constants default to UTF-8 and have a
070 * "_UTF_8" suffix. To get a version without a character set, use {@link #withoutParameters}.
071 *
072 * @since 12.0
073 * @author Gregory Kick
074 */
075@Beta
076@GwtCompatible
077@Immutable
078public final class MediaType {
079  private static final String CHARSET_ATTRIBUTE = "charset";
080  private static final ImmutableListMultimap<String, String> UTF_8_CONSTANT_PARAMETERS =
081      ImmutableListMultimap.of(CHARSET_ATTRIBUTE, Ascii.toLowerCase(UTF_8.name()));

087          .and(CharMatcher.isNot(' '))
088          .and(CharMatcher.noneOf("()<>@,;:\\\"/[]?="));

107  private static final Map<MediaType, MediaType> KNOWN_TYPES = Maps.newHashMap();
108
109  private static MediaType createConstant(String type, String subtype) {
110    MediaType mediaType =
111        addKnownType(new MediaType(type, subtype, ImmutableListMultimap.<String, String>of()));
112    mediaType.parsedCharset = Optional.absent();
113    return mediaType;
114  }
115
116  private static MediaType createConstantUtf8(String type, String subtype) {
117    MediaType mediaType = addKnownType(new MediaType(type, subtype, UTF_8_CONSTANT_PARAMETERS));
118    mediaType.parsedCharset = Optional.of(UTF_8);
119    return mediaType;
120  }
121
122  private static MediaType addKnownType(MediaType mediaType) {
123    KNOWN_TYPES.put(mediaType, mediaType);
124    return mediaType;
125  }

703  /**
704   * Returns an optional charset for the value of the charset parameter if it is specified.
705   *
706   * @throws IllegalStateException if multiple charset values have been set for this media type
707   * @throws IllegalCharsetNameException if a charset value is present, but illegal
708   * @throws UnsupportedCharsetException if a charset value is present, but no support is available
709   *     in this instance of the Java virtual machine
710   */
711  public Optional<Charset> charset() {
712    // racy single-check idiom, this is safe because Optional is immutable.
713    Optional<Charset> local = parsedCharset;
714    if (local == null) {
715      String value = null;
716      local = Optional.absent();
717      for (String currentValue : parameters.get(CHARSET_ATTRIBUTE)) {
718        if (value == null) {
719          value = currentValue;
720          local = Optional.of(Charset.forName(value));
721        } else if (!value.equals(currentValue)) {
722          throw new IllegalStateException(
723              "Multiple charset values defined: " + value + ", " + currentValue);
724        }
725      }
726      parsedCharset = local;
727    }
728    return local;
729  }

748  /**
749   * <em>Replaces</em> all parameters with the given attribute with parameters using the given
750   * values. If there are no values, any existing parameters with the given attribute are removed.
751   *
752   * @throws IllegalArgumentException if either {@code attribute} or {@code values} is invalid
753   * @since 24.0
754   */
755  public MediaType withParameters(String attribute, Iterable<String> values) {
756    checkNotNull(attribute);
757    checkNotNull(values);
758    String normalizedAttribute = normalizeToken(attribute);
759    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
760    for (Entry<String, String> entry : parameters.entries()) {
761      String key = entry.getKey();
762      if (!normalizedAttribute.equals(key)) {
763        builder.put(key, entry.getValue());
764      }
765    }
766    for (String value : values) {
767      builder.put(normalizedAttribute, normalizeParameterValue(normalizedAttribute, value));
768    }
769    MediaType mediaType = new MediaType(type, subtype, builder.build());
770    // if the attribute isn't charset, we can just inherit the current parsedCharset
771    if (!normalizedAttribute.equals(CHARSET_ATTRIBUTE)) {
772      mediaType.parsedCharset = this.parsedCharset;
773    }
774    // Return one of the constants if the media type is a known type.
775    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
776  }

793   * charset} parameter will be present on the new instance regardless of the number set on this

812  /**
813   * Returns {@code true} if this instance falls within the range (as defined by <a
814   * href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html">the HTTP Accept header</a>) given
815   * by the argument according to three criteria:
816   *
817   * <ol>
818   *   <li>The type of the argument is the wildcard or equal to the type of this instance.
819   *   <li>The subtype of the argument is the wildcard or equal to the subtype of this instance.
820   *   <li>All of the parameters present in the argument are present in this instance.
821   * </ol>
822   *
823   * <p>For example:
824   *
825   * <pre>{@code
826   * PLAIN_TEXT_UTF_8.is(PLAIN_TEXT_UTF_8) // true
827   * PLAIN_TEXT_UTF_8.is(HTML_UTF_8) // false
828   * PLAIN_TEXT_UTF_8.is(ANY_TYPE) // true
829   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE) // true
830   * PLAIN_TEXT_UTF_8.is(ANY_IMAGE_TYPE) // false
831   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_8)) // true
832   * PLAIN_TEXT_UTF_8.withoutParameters().is(ANY_TEXT_TYPE.withCharset(UTF_8)) // false
833   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_16)) // false
834   * }</pre>
835   *
836   * <p>Note that while it is possible to have the same parameter declared multiple times within a
837   * media type this method does not consider the number of occurrences of a parameter. For example,
838   * {@code "text/plain; charset=UTF-8"} satisfies {@code "text/plain; charset=UTF-8;
839   * charset=UTF-8"}.
840   */
841  public boolean is(MediaType mediaTypeRange) {
842    return (mediaTypeRange.type.equals(WILDCARD) || mediaTypeRange.type.equals(this.type))
843        && (mediaTypeRange.subtype.equals(WILDCARD) || mediaTypeRange.subtype.equals(this.subtype))
844        && this.parameters.entries().containsAll(mediaTypeRange.parameters.entries());
845  }

848   * Creates a new media type with the given type and subtype.
849   *
850   * @throws IllegalArgumentException if type or subtype is invalid or if a wildcard is used for the
851   *     type, but not the subtype.
852   */
853  public static MediaType create(String type, String subtype) {
854    MediaType mediaType = create(type, subtype, ImmutableListMultimap.<String, String>of());
855    mediaType.parsedCharset = Optional.absent();
856    return mediaType;
857  }

859  private static MediaType create(
860      String type, String subtype, Multimap<String, String> parameters) {
861    checkNotNull(type);
862    checkNotNull(subtype);
863    checkNotNull(parameters);
864    String normalizedType = normalizeToken(type);
865    String normalizedSubtype = normalizeToken(subtype);
866    checkArgument(
867        !WILDCARD.equals(normalizedType) || WILDCARD.equals(normalizedSubtype),
868        "A wildcard type cannot be used with a non-wildcard subtype");
869    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
870    for (Entry<String, String> entry : parameters.entries()) {
871      String attribute = normalizeToken(entry.getKey());
872      builder.put(attribute, normalizeParameterValue(attribute, entry.getValue()));
873    }
874    MediaType mediaType = new MediaType(normalizedType, normalizedSubtype, builder.build());
875    // Return one of the constants if the media type is a known type.
876    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
877  }

924  private static String normalizeToken(String token) {

927  }

933  /**
934   * Parses a media type from its string representation.
935   *
936   * @throws IllegalArgumentException if the input is not parsable
937   */
938  public static MediaType parse(String input) {
939    checkNotNull(input);
940    Tokenizer tokenizer = new Tokenizer(input);
941    try {
942      String type = tokenizer.consumeToken(TOKEN_MATCHER);
943      tokenizer.consumeCharacter('/');
944      String subtype = tokenizer.consumeToken(TOKEN_MATCHER);
945      ImmutableListMultimap.Builder<String, String> parameters = ImmutableListMultimap.builder();
946      while (tokenizer.hasMore()) {
947        tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
948        tokenizer.consumeCharacter(';');
949        tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
950        String attribute = tokenizer.consumeToken(TOKEN_MATCHER);
951        tokenizer.consumeCharacter('=');
952        final String value;
953        if ('"' == tokenizer.previewChar()) {
954          tokenizer.consumeCharacter('"');
955          StringBuilder valueBuilder = new StringBuilder();
956          while ('"' != tokenizer.previewChar()) {
957            if ('\\' == tokenizer.previewChar()) {
958              tokenizer.consumeCharacter('\\');
959              valueBuilder.append(tokenizer.consumeCharacter(ascii()));
960            } else {
961              valueBuilder.append(tokenizer.consumeToken(QUOTED_TEXT_MATCHER));
962            }
963          }
964          value = valueBuilder.toString();
965          tokenizer.consumeCharacter('"');
966        } else {
967          value = tokenizer.consumeToken(TOKEN_MATCHER);
968        }
969        parameters.put(attribute, value);
970      }
971      return create(type, subtype, parameters.build());
972    } catch (IllegalStateException e) {
973      throw new IllegalArgumentException("Could not parse '" + input + "'", e);
974    }
975  }

977  private static final class Tokenizer {
978    final String input;
979    int position = 0;
980
981    Tokenizer(String input) {
982      this.input = input;
983    }
984
985    String consumeTokenIfPresent(CharMatcher matcher) {
986      checkState(hasMore());
987      int startPosition = position;
988      position = matcher.negate().indexIn(input, startPosition);
989      return hasMore() ? input.substring(startPosition, position) : input.substring(startPosition);
990    }
991
992    String consumeToken(CharMatcher matcher) {
993      int startPosition = position;
994      String token = consumeTokenIfPresent(matcher);
995      checkState(position != startPosition);
996      return token;
997    }
998
999    char consumeCharacter(CharMatcher matcher) {
1000      checkState(hasMore());
1001      char c = previewChar();
1002      checkState(matcher.matches(c));
1003      position++;
1004      return c;
1005    }
1006
1007    char consumeCharacter(char c) {
1008      checkState(hasMore());
1009      checkState(previewChar() == c);
1010      position++;
1011      return c;
1012    }
1013
1014    char previewChar() {
1015      checkState(hasMore());
1016      return input.charAt(position);
1017    }
1018
1019    boolean hasMore() {
1020      return (position >= 0) && (position < input.length());
1021    }
1022  }

1024  @Override
1025  public boolean equals(@Nullable Object obj) {
1026    if (obj == this) {
1027      return true;
1028    } else if (obj instanceof MediaType) {
1029      MediaType that = (MediaType) obj;
1030      return this.type.equals(that.type)
1031          && this.subtype.equals(that.subtype)
1032          // compare parameters regardless of order
1033          && this.parametersAsMap().equals(that.parametersAsMap());
1034    } else {
1035      return false;
1036    }
1037  }

1050  private static final MapJoiner PARAMETER_JOINER = Joiner.on("; ").withKeyValueSeparator("=");
1051
1052  /**
1053   * Returns the string representation of this media type in the format described in <a
1054   * href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>.
1055   */
1056  @Override
1057  public String toString() {
1058    // racy single-check idiom, safe because String is immutable
1059    String result = toString;
1060    if (result == null) {
1061      result = computeToString();
1062      toString = result;
1063    }
1064    return result;
1065  }
1066
1067  private String computeToString() {
1068    StringBuilder builder = new StringBuilder().append(type).append('/').append(subtype);
1069    if (!parameters.isEmpty()) {
1070      builder.append("; ");
1071      Multimap<String, String> quotedParameters =
1072          Multimaps.transformValues(
1073              parameters,
1074              new Function<String, String>() {
1075                @Override
1076                public String apply(String value) {
1077                  return TOKEN_MATCHER.matchesAllOf(value) ? value : escapeAndQuote(value);
1078                }
1079              });
1080      PARAMETER_JOINER.appendTo(builder, quotedParameters.entries());
1081    }
1082    return builder.toString();
1083  }
1084
1085  private static String escapeAndQuote(String value) {
1086    StringBuilder escaped = new StringBuilder(value.length() + 16).append('"');
1087    for (int i = 0; i < value.length(); i++) {
1088      char ch = value.charAt(i);
1089      if (ch == '\r' || ch == '\\' || ch == '"') {
1090        escaped.append('\\');
1091      }
1092      escaped.append(ch);
1093    }
1094    return escaped.append('"').toString();
1095  }
