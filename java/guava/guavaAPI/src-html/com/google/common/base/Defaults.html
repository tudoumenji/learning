015package com.google.common.base;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018
019import com.google.common.annotations.GwtIncompatible;
020import org.checkerframework.checker.nullness.qual.Nullable;
021
022/**
023 * This class provides default values for all Java types, as defined by the JLS.
024 *
025 * @author Ben Yu
026 * @since 1.0
027 */
028@GwtIncompatible
029public final class Defaults {
030  private Defaults() {}
031
032  private static final Double DOUBLE_DEFAULT = Double.valueOf(0d);
033  private static final Float FLOAT_DEFAULT = Float.valueOf(0f);
034
035  /**
036   * Returns the default value of {@code type} as defined by JLS --- {@code 0} for numbers, {@code
037   * false} for {@code boolean} and {@code '\0'} for {@code char}. For non-primitive types and
038   * {@code void}, {@code null} is returned.
039   */
040  @SuppressWarnings("unchecked")
041  public static <T> @Nullable T defaultValue(Class<T> type) {
042    checkNotNull(type);
043    if (type == boolean.class) {
044      return (T) Boolean.FALSE;
045    } else if (type == char.class) {
046      return (T) Character.valueOf('\0');
047    } else if (type == byte.class) {
048      return (T) Byte.valueOf((byte) 0);
049    } else if (type == short.class) {
050      return (T) Short.valueOf((short) 0);
051    } else if (type == int.class) {
052      return (T) Integer.valueOf(0);
053    } else if (type == long.class) {
054      return (T) Long.valueOf(0L);
055    } else if (type == float.class) {
056      return (T) FLOAT_DEFAULT;
057    } else if (type == double.class) {
058      return (T) DOUBLE_DEFAULT;
059    } else {
060      return null;
061    }
062  }
063}
